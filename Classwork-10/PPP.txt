------------------------------------------------------------
MAIN PROGRAM - Login Loop
------------------------------------------------------------

PRINT "School Management System"

# Keep asking until valid credentials are entered (unlimited attempts)
REPEAT
    READ username
    READ password

    IF username IN users AND users[username][password] = password THEN

        name ← users[username][name]
        rol  ← users[username][rol]
        PRINT "Welcome, " + name + " (" + rol + ")"

        # ── STUDENT ──────────────────────────────────────────
        IF rol = 'student' THEN

            grades ← notes[username]

            # Build passed set (grade >= 7.0) and pending (set difference)
            passed  ← { s FOR EACH s IN subjects IF grades[s] >= 7.0 }
            pending ← set(subjects) - passed

            # Print table with only passed subjects
            PRINT "Report Card: " + name
            PRINT table header row
            FOR EACH subject IN subjects DO
                IF subject IN passed THEN
                    PRINT subject + grades[subject] as table row
                END IF
            END FOR
            PRINT table footer

            # Print passed count and list
            PRINT "Passed subjects (" + count(passed) + " of " + count(subjects) + "):"
            FOR EACH s IN subjects DO
                IF s IN passed THEN
                    PRINT "[+] " + s
                END IF
            END FOR

            # Print pending list
            IF pending is not empty THEN
                PRINT "Pending subjects (" + count(pending) + "):"
                FOR EACH s IN subjects DO
                    IF s IN pending THEN
                        PRINT "[-] " + s
                    END IF
                END FOR
            ELSE
                PRINT "All subjects passed!"
            END IF

        # ── PROFESSOR ─────────────────────────────────────────
        ELSE IF rol = 'professor' THEN

            # Print all students filtered by role
            FOR EACH user, data IN users DO
                IF data[rol] = 'student' THEN
                    PRINT user + data[name]
                END IF
            END FOR

            # Loop until professor types 'exit'
            REPEAT
                READ student

                IF student = 'exit' THEN
                    EXIT LOOP
                END IF

                IF student NOT IN notes THEN
                    PRINT "Not found. Try again."
                    CONTINUE
                END IF

                # Print current grades as table
                FOR EACH subject IN subjects DO
                    PRINT subject + notes[student][subject] as table row
                END FOR

                READ subject
                IF subject NOT IN subjects THEN
                    PRINT "Invalid. Try again."
                    CONTINUE
                END IF

                READ new_grade

                # Ask for confirmation before saving
                PRINT "Change " + subject + " from " + current + " to " + new_grade + "?"
                READ confirm

                IF confirm = 'yes' THEN
                    notes[student][subject] ← new_grade
                    PRINT "Grade updated successfully."
                ELSE
                    PRINT "Change cancelled."
                END IF

            UNTIL student = 'exit'

        # ── COORDINATOR ───────────────────────────────────────
        ELSE IF rol = 'coordinator' THEN

            # Print professors filtered by role
            FOR EACH user, data IN users DO
                IF data[rol] = 'professor' THEN
                    PRINT user + data[name]
                END IF
            END FOR

            # Print subjects from tuple
            FOR EACH subject IN subjects DO
                PRINT subject
            END FOR

            # Print ONE combined table: students as rows, subjects as columns
            PRINT table header (Student | subject1 | subject2 | ...)
            FOR EACH student, grades IN notes DO
                PRINT student name + all grades as one row
            END FOR
            PRINT table footer

        END IF

        EXIT LOGIN LOOP

    ELSE
        PRINT "Invalid username or password. Please try again."
    END IF

UNTIL login is successful
